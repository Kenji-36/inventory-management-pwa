```typescript
import { create } from 'zustand';
import { devtools, persist } from 'zustand/middleware';

interface BlogProject {
  theme: string;
  currentStep: number;
  // ...
}

interface BlogStore {
  project: BlogProject | null;
  setTheme: (theme: string) => void;
  nextStep: () => void;
  // ...
}

export const useBlogStore = create<BlogStore>()(
  devtools(
    persist(
      (set) => ({
        project: null,
        setTheme: (theme) => set((state) => ({
          project: { ...state.project!, theme }
        })),
        nextStep: () => set((state) => ({
          project: { ...state.project!, currentStep: state.project!.currentStep + 1 }
        })),
      }),
      { name: 'blog-generator' }
    )
  )
);
```

```typescript
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

const themeSchema = z.object({
  theme: z.string().min(1, '必須項目です').max(500, '500文字以内で入力してください'),
});

type ThemeFormData = z.infer<typeof themeSchema>;

export function ThemeInput() {
  const { register, handleSubmit, formState: { errors } } = useForm<ThemeFormData>({
    resolver: zodResolver(themeSchema),
  });

  const onSubmit = (data: ThemeFormData) => {
    console.log(data);
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <textarea {...register('theme')} />
      {errors.theme && <p>{errors.theme.message}</p>}
      <button type="submit">送信</button>
    </form>
  );
}
```

2. **状態管理**: loading, error, dataを自動管理

**使用例**:
```typescript
import { useQuery, useMutation } from '@tanstack/react-query';

export function useGenerateHeadlines(theme: string) {
  return useQuery({
    queryKey: ['headlines', theme],
    queryFn: async () => {
      const res = await fetch('/api/generate/headline', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ theme }),
      });
      if (!res.ok) throw new Error('生成に失敗しました');
      return res.json();
    },
    enabled: !!theme,
    retry: 2,
    staleTime: 5 * 60 * 1000, // 5分
  });
}

export function useGenerateContent() {
  return useMutation({
    mutationFn: async (data: ContentGenerationRequest) => {
      const res = await fetch('/api/generate/content', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data),
      });
      return res.json();
    },
  });
}
```
